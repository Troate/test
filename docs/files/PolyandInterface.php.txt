interface Button
{
    public function pressButton();
}

class PushButton implements Button
{
    public function pressButton()
    {
        echo "Pushed\n";
        return "Pushed";
    }
}

class PullDown implements Button
{
    public function pressButton() {
        echo "Pulled\n";
        return "Pulled";
    }
}
